 18 /**
 19  * Element
 20  *
 21  * @package ILess\Node
 22  */
 23 class ElementNode extends Node
 24 {
 25     /**
 26      * Node type
 27      *
 28      * @var string
 29      */
 30     protected $type = 'Element';
 31 
 32     /**
 33      * ILess\Node combinator
 34      *
 35      * @var CombinatorNode
 36      */
 37     public $combinator;
 38 
 39     /**
 40      * The current index
 41      *
 42      * @var integer
 43      */
 44     public $index = 0;
 45 
 46     /**
 47      * Constructor
 48      *
 49      * @param CombinatorNode|string $combinator The combinator
 50      * @param string|Node $value The value
 51      * @param integer $index The current index
 52      * @param FileInfo $currentFileInfo Current file information
 53      */
 54     public function __construct($combinator, $value, $index = 0, FileInfo $currentFileInfo = null)
 55     {
 56         $this->combinator = $combinator instanceof CombinatorNode ? $combinator : new CombinatorNode($combinator);
 57 
 58         if (is_string($value)) {
 59             $value = trim($value);
 60         } elseif (!$value) {
 61             $value = '';
 62         }
 63 
 64         parent::__construct($value);
 65 
 66         $this->index = $index;
 67         $this->currentFileInfo = $currentFileInfo;
 68     }
 69 
 70     /**
 71      * @inheritdoc
 72      */
 73     public function accept(VisitorInterface $visitor)
 74     {
 75         $this->combinator = $visitor->visit($this->combinator);
 76         if (is_object($this->value)) {
 77             $this->value = $visitor->visit($this->value);
 78         }
 79     }
 80 
 81     /**
 82      * @inheritdoc
 83      */
 84     public function generateCSS(Context $context, OutputInterface $output)
 85     {
 86         $output->add($this->toCSS($context), $this->currentFileInfo, $this->index);
 87     }
 88 
 89     /**
 90      * Convert to CSS
 91      *
 92      * @param Context $context The context
 93      * @return string
 94      */
 95     public function toCSS(Context $context)
 96     {
 97         $firstSelector = $context->firstSelector;
 98         if ($this->value instanceof ParenNode) {
 99             $context->firstSelector = true;
100         }
101 
102         $value = $this->value instanceof GenerateCSSInterface ? $this->value->toCSS($context) : $this->value;
103 
104         $context->firstSelector = $firstSelector;
105         if ($value === '' && strlen($this->combinator->value) && $this->combinator->value[0] === '&') {
106             return '';
107         } else {
108             return $this->combinator->toCSS($context).$value;
109         }
110     }
111 
112     /**
113      * Compiles the node
114      *
115      * @param Context $context The context
116      * @param array|null $arguments Array of arguments
117      * @param boolean|null $important Important flag
118      * @return ElementNode
119      */
120     public function compile(Context $context, $arguments = null, $important = null)
121     {
122         return new ElementNode($this->combinator,
123             $this->value instanceof CompilableInterface ? $this->value->compile($context) : $this->value,
124             $this->index, $this->currentFileInfo
125         );
126     }
127 
128 }
