 82     /**
 83      * Constructor
 84      *
 85      * @param array|string $rgb The RGB components as an array or string definition
 86      * @param integer $alpha The alpha channel
 87      * @param string $originalForm
 88      */
 89     public function __construct($rgb = array(255, 255, 255), $alpha = 1, $originalForm = null)
 90     {
 91         if (is_array($rgb)) {
 92             $this->rgb = $rgb;
 93         } // string
 94         else {
 95 
 96             // this is a named color
 97             if ($color = self::color($rgb)) {
 98                 $this->keyword = $rgb;
 99                 $rgb = $color;
100             }
101 
102             // strip #
103             $rgb = trim($rgb, '#');
104             if (strlen($rgb) == 6) {
105                 foreach (str_split($rgb, 2) as $c) {
106                     $this->rgb[] = hexdec($c);
107                 }
108             } elseif (strlen($rgb) == 3) {
109                 $this->short = true;
110                 foreach (str_split($rgb, 1) as $c) {
111                     $this->rgb[] = hexdec($c.$c);
112                 }
113             } elseif (strtolower($rgb) == 'transparent') {
114                 $this->rgb = array(255, 255, 255);
115                 $this->isTransparentKeyword = true;
116                 $alpha = 0;
117             } else {
118                 throw new InvalidArgumentException('Argument must be a color keyword or 3/6 digit hex e.g. #FFF.');
119             }
120         }
121 
122         $this->originalForm = $originalForm;
123         // limit alpha channel
124         $this->alpha = is_numeric($alpha) ? $alpha : 1;
125     }

252     /**
253      * Returns the luma
254      *
255      * @return integer
256      */
257     public function getLuma()
258     {
259         if ($this->luma !== null) {
260             return $this->luma;
261         }
262 
263         $r = $this->rgb[0] / 255;
264         $g = $this->rgb[1] / 255;
265         $b = $this->rgb[2] / 255;
266 
267         $r = ($r <= 0.03928) ? $r / 12.92 : pow((($r + 0.055) / 1.055), 2.4);
268         $g = ($g <= 0.03928) ? $g / 12.92 : pow((($g + 0.055) / 1.055), 2.4);
269         $b = ($b <= 0.03928) ? $b / 12.92 : pow((($b + 0.055) / 1.055), 2.4);
270 
271         $this->luma = 0.2126 * $r + 0.7152 * $g + 0.0722 * $b;
272 
273         return $this->luma;
274     }

293     /**
294      * Converts to HSL
295      *
296      * @return array
297      */
298     public function toHSL()
299     {
300         if ($this->hsl) {
301             return $this->hsl;
302         }
303 
304         $r = $this->rgb[0] / 255;
305         $g = $this->rgb[1] / 255;
306         $b = $this->rgb[2] / 255;
307         $a = $this->alpha;
308 
309         $max = max($r, $g, $b);
310         $min = min($r, $g, $b);
311         $l = ($max + $min) / 2;
312         $d = $max - $min;
313 
314         if ($max === $min) {
315             $h = $s = 0;
316         } else {
317             $s = $l > 0.5 ? $d / (2 - $max - $min) : $d / ($max + $min);
318 
319             switch ($max) {
320                 case $r:
321                     $h = ($g - $b) / $d + ($g < $b ? 6 : 0);
322                     break;
323                 case $g:
324                     $h = ($b - $r) / $d + 2;
325                     break;
326                 case $b:
327                     $h = ($r - $g) / $d + 4;
328                     break;
329             }
330             $h /= 6;
331         }
332 
333         $this->hsl = array('h' => $h * 360, 's' => $s, 'l' => $l, 'a' => $a);
334 
335         return $this->hsl;
336     }

338     /**
339      * Converts to HSV
340      *
341      * @return array
342      */
343     public function toHSV()
344     {
345         if ($this->hsv) {
346             return $this->hsv;
347         }
348 
349         $r = $this->rgb[0] / 255;
350         $g = $this->rgb[1] / 255;
351         $b = $this->rgb[2] / 255;
352         $a = $this->alpha;
353 
354         $max = max($r, $g, $b);
355         $min = min($r, $g, $b);
356 
357         $v = $max;
358 
359         $d = $max - $min;
360         if ($max === 0) {
361             $s = 0;
362         } else {
363             $s = $d / $max;
364         }
365 
366         if ($max === $min) {
367             $h = 0;
368         } else {
369             switch ($max) {
370                 case $r:
371                     $h = ($g - $b) / $d + ($g < $b ? 6 : 0);
372                     break;
373                 case $g:
374                     $h = ($b - $r) / $d + 2;
375                     break;
376                 case $b:
377                     $h = ($r - $g) / $d + 4;
378                     break;
379             }
380             $h /= 6;
381         }
382 
383         return array('h' => $h * 360, 's' => $s, 'v' => $v, 'a' => $a);
384     }

407     private function toHex($rgb)
408     {
409         $parts = array_map(

426     /**
427      * Returns the color as HEX string (when transparency present, in RGBA model)
428      *
429      * @param boolean $compress Compress the color?
430      * @param boolean $canShorten Can the color be shortened if possible?
431      * @return string
432      */
433     public function toString($compress = false, $canShorten = false)
434     {
435         if ($this->isTransparentKeyword) {
436             return 'transparent';
437         }
438 
439         if ($this->originalForm) {
440             return $this->originalForm;
441         }
442 
443         $alpha = Math::toFixed($this->alpha + 2e-16, 8);
444 
445         if ($alpha < 1) {
446             $fixedRGB = $this->getFixedRGB();
447 
448             return sprintf(
449                 'rgba(%s)',
450                 join(
451                     $compress ? ',' : ', ',
452                     array(
453                         $fixedRGB[0],
454                         $fixedRGB[1],
455                         $fixedRGB[2],
456                         Math::clean($this->clamp($alpha, 1)),
457                     )
458                 )
459             );
460         }
461 
462         // prevent named colors
463         if ($this->keyword) {
464             return $this->keyword;
465         }
466 
467         $color = array();
468         foreach ($this->getFixedRgb() as $i) {
469             $color[] = str_pad(dechex(Math::round($i)), 2, '0', STR_PAD_LEFT);
470         }
471 
472         $color = join('', $color);
473 
474         // convert color to short format
475         if ($canShorten && $color[0] === $color[1] && $color[2] === $color[3] && $color[4] === $color[5]) {
476             $color = $color[0].$color[2].$color[4];
477         }
478 
479         $color = '#'.$color;
480 
481 
482         return $color;
483     }
