241     /**
242      * Runs the task based on the arguments
243      *
244      * @return true|integer True on success, error code on failure
245      */
246     public function run()
247     {
248         if (!$this->isValid()) {
249             echo $this->getUsage();
250 
251             // return error
252             return 1;
253         } elseif ($this->getOption('version')) {
254             echo Parser\Core::VERSION.' [compatible with less.js '.Parser\Core::LESS_JS_VERSION.']'.PHP_EOL;
255 
256             return 0;
257         } elseif ($this->getOption('help')) {
258             echo $this->getUsage();
259 
260             return 0;
261         }
262 
263         try {
264             $parser = new Parser($this->prepareOptionsForTheParser());
265 
266             $toBeParsed = $this->cliArguments['arguments'][0];
267 
268             // read from stdin
269             if (in_array($toBeParsed, $this->stdAliases)) {
270                 $content = file_get_contents('php://stdin');
271                 $parser->parseString($content);
272             } else {
273                 if (!Util::isPathAbsolute($toBeParsed)) {
274                     $toBeParsed = sprintf('%s/%s', $this->currentDir, $toBeParsed);
275                 }
276                 $parser->parseFile($toBeParsed);
277             }
278 
279             $toBeSavedTo = null;
280             if (isset($this->cliArguments['arguments'][1])) {
281                 $toBeSavedTo = $this->cliArguments['arguments'][1];
282                 if (!Util::isPathAbsolute($toBeSavedTo)) {
283                     $toBeSavedTo = sprintf('%s/%s', $this->currentDir, $toBeSavedTo);
284                 }
285             }
286 
287             $css = $parser->getCSS();
288 
289             // where to put the css?
290             if ($toBeSavedTo) {
291                 // write the result
292                 $this->saveCSS($toBeSavedTo, $css, $this->getOption('append'));
293             } else {
294                 echo $css;
295             }
296         } catch (Exception $e) {
297             if (!$this->getOption('silent')) {
298                 $this->renderException($e);
299             }
300 
301             return $e->getCode();
302         }
303 
304         return true;
305     }

307     /**
308      * Prepares options for the parser
309      *
310      * @return array
311      */
312     protected function prepareOptionsForTheParser()
313     {
314         $options = [];
315 
316         foreach ($this->getOptions() as $option => $value) {
317             switch ($option) {
318                 case 'source_map':
319 
320                     $options['source_map'] = true;
321                     $options['source_map_options'] = [];
322 
323                     if (is_string($value)) {
324                         $options['source_map_options']['write_to'] = $value;
325                     }
326 
327                     if ($basePath = $this->getOption('source_map_base_path')) {
328                         $options['source_map_options']['base_path'] = $basePath;
329                     }
330 
331                     if ($url = $this->getOption('source_map_url')) {
332                         $options['source_map_options']['url'] = $url;
333                     } // same as write to
334                     elseif (is_string($value)) {
335                         $options['source_map_options']['url'] = basename($value);
336                     }
337 
338                     break;
339 
340                 // skip options which are processed above or invalid
341                 case 'source_map_base_path':
342                 case 'source_map_url':
343                 case 'silent':
344                 case 'no_color':
345                 case 'append':
346                     continue 2;
347 
348                 case 'no_ie_compat':
349                     $options['ie_compat'] = false;
350                     continue 2;
351 
352                 // less.js compatibility options
353                 case 'line_numbers':
354                     $options['dump_line_numbers'] = $value;
355                     break;
356 
357                 default:
358                     $options[$option] = $value;
359                     break;
360             }
361 
362             // all is passed, The context checks if the option is valid
363             $options[$option] = $value;
364         }
365 
366         return $options;
367     }

385     /**
386      * Returns the ILess\CLI usage
387      *
388      * @return string
389      */
390     public function getUsage()
391     {
392         $options = [];
393         $max = 0;
394         foreach ($this->validOptions as $optionName => $properties) {
395             $optionName = str_replace('_', '-', $optionName);
396             list($help, $flags) = $properties;
397             if ($flags) {
398                 $option = sprintf('  -%s, --%s', join(',-', $flags), $optionName);
399             } else {
400                 $option = sprintf('  --%s', $optionName);
401             }
402 
403             // find the largest line
404             if ((strlen($option) + 2) > $max) {
405                 $max = strlen($option) + 2;
406             }
407 
408             $options[] = [
409                 $option,
410                 $help,
411             ];
412         }
413 
414         $optionsFormatted = [];
415         foreach ($options as $option) {
416             list($name, $help) = $option;
417             // line will be too long
418             if (strlen($name.$help) + 2 > self::MAX_LINE_LENGTH) {
419                 $help = wordwrap($help, self::MAX_LINE_LENGTH, PHP_EOL.str_repeat(' ', $max + 2));
420             }
421             $optionsFormatted[] = sprintf(' %-'.$max.'s %s', $name, $help);
422         }
423 
424         return strtr('
425 {%signature}
426 
427 usage: {%script_name} [option option=parameter ...] source [destination]
428 
429 If source is set to `-` (dash or hyphen-minus), input is read from stdin.
430 
431 options:
432 {%options}'.PHP_EOL, [
433             '{%signature}' => $this->getSignature(),
434             '{%script_name}' => $this->scriptName,
435             '{%options}' => join(PHP_EOL, $optionsFormatted),
436         ]);
437     }

514     /**
515      * Parses the $argv array to a more useful array
516      *
517      * @param array $args The $argv array
518      * @return array
519      * @link http://php.net/manual/en/features.commandline.php#83843
520      */
521     protected function parseArguments($args)
522     {
523         $return = [
524             'arguments' => [],
525             'flags' => [],
526             'options' => [],
527         ];
528 
529         while ($arg = array_shift($args)) {
530             if (in_array($arg, $this->stdAliases)) {
531                 $return['arguments'][] = $arg;
532             } // Is it a command? (prefixed with --)
533             elseif (substr($arg, 0, 2) === '--') {
534                 $value = '';
535                 $command = substr($arg, 2);
536                 // is it the syntax '--option=argument'?
537                 if (strpos($command, '=') !== false) {
538                     list($command, $value) = explode('=', $command, 2);
539                 }
540                 $return['options'][$command] = !empty($value) ? $this->convertValue($value) : true;
541             } // Is it a flag or a serial of flags? (prefixed with -)
542             else {
543                 if (substr($arg, 0, 1) === '-') {
544                     for ($i = 1; isset($arg[$i]); $i++) {
545                         $return['flags'][] = $arg[$i];
546                     }
547                 } else {
548                     $return['arguments'][] = $arg;
549                 }
550             }
551         }
552 
553         return $return;
554     }

556     /**
557      * Converts the value. Parses strings like "false" to boolean false,
558      * "true" to boolean true
559      *
560      * @param string $value
561      * @return mixed
562      */
563     protected function convertValue($value)
564     {
565         switch (strtolower($value)) {
566             case '0':
567             case 'false':
568             case 'no':
569                 $value = false;
570                 break;
571 
572             case '1':
573             case 'true':
574             case 'yes':
575                 $value = true;
576                 break;
577         }
578 
579         return $value;
