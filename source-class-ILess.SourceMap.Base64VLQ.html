197     /**
198      * Convert from a two-complement value to a value where the sign bit is
199      * is placed in the least significant bit.  For example, as decimals:
200      *   1 becomes 2 (10 binary), -1 becomes 3 (11 binary)
201      *   2 becomes 4 (100 binary), -2 becomes 5 (101 binary)
202      * We generate the value for 32 bit machines, hence -2147483648 becomes 1, not 4294967297,
203      * even on a 64 bit machine.
204      */
205     public function toVLQSigned($aValue)
206     {
207         return 0xffffffff & ($aValue < 0 ? ((-$aValue) << 1) + 1 : ($aValue << 1) + 0);
208     }
209 
210     /**
211      * Convert to a two-complement value from a value where the sign bit is
212      * is placed in the least significant bit. For example, as decimals:
213      *   2 (10 binary) becomes 1, 3 (11 binary) becomes -1
214      *   4 (100 binary) becomes 2, 5 (101 binary) becomes -2
215      * We assume that the value was generated with a 32 bit machine in mind.
216      * Hence
217      *   1 becomes -2147483648
218      * even on a 64 bit machine.
219      */
220     public function fromVLQSigned($aValue)
221     {
222         return $aValue & 1 ? $this->zeroFill(~$aValue + 2, 1) | (-1 - 0x7fffffff) : $this->zeroFill($aValue, 1);
223     }

225     /**
226      * Return the base 64 VLQ encoded value.
227      *
228      * @param string $aValue The value to encode
229      *
230      * @return string The encoded value
231      */
232     public function encode($aValue)
233     {
234         $encoded = '';
235         $vlq = $this->toVLQSigned($aValue);
236         do {
237             $digit = $vlq & $this->mask;
238             $vlq = $this->zeroFill($vlq, $this->shift);
239             if ($vlq > 0) {
240                 $digit |= $this->continuationBit;
241             }
242             $encoded .= $this->base64Encode($digit);
243         } while ($vlq > 0);
244 
245         return $encoded;
246     }
247 
248     /**
249      * Return the value decoded from base 64 VLQ.
250      *
251      * @param string $encoded The encoded value to decode
252      *
253      * @return string The decoded value
254      */
255     public function decode($encoded)
256     {
257         $vlq = 0;
258         $i = 0;
259         do {
260             $digit = $this->base64Decode($encoded[$i]);
261             $vlq |= ($digit & $this->mask) << ($i * $this->shift);
262             ++$i;
263         } while ($digit & $this->continuationBit);
264 
265         return $this->fromVLQSigned($vlq);
266     }

273      *
274      * @return int
275      */
276     public function zeroFill($a, $b)
277     {
278         return ($a >= 0) ? ($a >> $b) : ($a >> $b) & (PHP_INT_MAX >> ($b - 1));
